class TrainingDataCollector {
  constructor(videoCapture, gpsTracker, storage) {
    this.videoCapture = videoCapture;
    this.gpsTracker = gpsTracker;
    this.storage = storage;
    this.collectionMode = false;
    this.collectionInterval = null;
  }
  
  // Enable data collection mode
  startCollecting(fps = 2) {
    this.collectionMode = true;
    const interval = 1000 / fps;
    
    this.collectionInterval = setInterval(async () => {
      const frame = await this.videoCapture.captureFrame();
      const metadata = {
        timestamp: Date.now(),
        gpsPosition: this.gpsTracker.getCurrentPosition(),
        distance: this.gpsTracker.getCumulativeDistance(),
        speed: this.gpsTracker.getCurrentSpeed(),
        filename: `img_${Date.now()}_${Math.random().toString(36).substr(2, 9)}.jpg`
      };
      
      await this.saveTrainingImage(frame, metadata);
    }, interval);
  }
  
  async saveTrainingImage(imageBlob, metadata) {
    // Save to IndexedDB or export folder
    await this.storage.saveTrainingData({
      image: imageBlob,
      metadata: metadata,
      annotated: false,
      exported: false
    });
  }
  
  stopCollecting() {
    if (this.collectionInterval) {
      clearInterval(this.collectionInterval);
      this.collectionMode = false;
    }
  }
  
  // Export collected images for annotation
  async exportForAnnotation() {
    const images = await this.storage.getTrainingData({ exported: false });
    
    // Create downloadable ZIP or folder structure
    // Format: /training_data/YYYY-MM-DD/images/

  }

```
### Annotation Guidelines

**Bounding Box Rules:**
```
✅ GOOD ANNOTATION:
- Tight boxes around entire object
- Include all visible parts
- No excessive empty space
- Consistent labeling across images

❌ BAD ANNOTATION:
- Box too loose or too tight
- Missing parts of object
- Inconsistent labeling
- Overlapping boxes for same object
Specific Object Guidelines:

Power Lines / Wires:

Box should encompass entire visible span
For multiple parallel lines, one box if close together
Separate boxes for clearly distinct lines
Label voltage level if distinguishable


Trees:

tree_branch: Only overhanging branches
tree_full: Entire tree visible including trunk
Box includes foliage, not just trunk


Bridges/Overpasses:

Include entire structure width
From edge to edge as visible
Include supports if visible


Signs:

Tight box around sign face
Include pole only if very short
Separate box for pole if needed


Traffic Lights:

Include housing and lights
All light heads in one box if same unit



Quality Metrics:

Annotation time: 20-40 seconds per image (with practice)
Quality check: Review 10% of annotations randomly
Inter-annotator agreement: >90% for consistent labeling

javascript// onnx-detection.js - ONNX Runtime Web implementation

import * as ort from 'onnxruntime-web';

class ONNXObjectDetection {
  constructor() {
    this.session = null;
    this.modelPath = '/models/best_quantized.onnx';
    this.inputShape = [1, 3, 640, 640];  // NCHW format
    this.classNames = [...];  // Load from classes.json
    this.numClasses = 26;
  }
  
  async loadModel() {
    try {
      // Configure ONNX Runtime
      ort.env.wasm.numThreads = 4;  // Use 4 threads
      ort.env.wasm.simd = true;     // Enable SIMD
      
      // Load model from cache
      this.session = await ort.InferenceSession.create(
        this.modelPath,
        {
          executionProviders: ['wasm'],  // WebAssembly backend
          graphOptimizationLevel: 'all'
        }
      );
      
      console.log('ONNX model loaded successfully');
      return true;
    } catch (error) {
      console.error('Model loading failed:', error);
      return false;
    }
  }
  
  async detect(imageData) {
    if (!this.session) return null;
    
    try {
      // Preprocess image
      const tensor = await this.preprocessImage(imageData);
      
      // Run inference
      const startTime = performance.now();
      const results = await this.session.run({ images: tensor });
      const inferenceTime = performance.now() - startTime;
      
      console.log(`Inference time: ${inferenceTime.toFixed(2)}ms`);
      
      // Post-process results
      const detections = this.postProcess(results);
      
      return {
        detections: detections,
        inferenceTime: inferenceTime
      };
    } catch (error) {
      console.error('Detection error:', error);
      return null;
    }
  }
  
  async preprocessImage(imageData) {
    // Convert image to tensor
    // 1. Resize to 640x640
    // 2. Normalize to [0, 1]
    // 3. Convert to NCHW format (batch, channels, height, width)
    
    const canvas = document.createElement('canvas');
    canvas.width = 640;
    canvas.height = 640;
    const ctx = canvas.getContext('2d');
    
    // Draw and resize image
    const img = await this.loadImage(imageData);
    ctx.drawImage(img, 0, 0, 640, 640);
    
    // Get pixel data
    const imageDataResized = ctx.getImageData(0, 0, 640, 640);
    const pixels = imageDataResized.data;
    
    // Convert to float array (NCHW format)
    const float32Data = new Float32Array(1 * 3 * 640 * 640);
    
    for (let i = 0; i < 640 * 640; i++) {
      float32Data[i] = pixels[i * 4] / 255.0;           // R
      float32Data[640 * 640 + i] = pixels[i * 4 + 1] / 255.0;  // G
      float32Data[640 * 640 * 2 + i] = pixels[i * 4 + 2] / 255.0;  // B
    }
    
    // Create tensor
    const tensor = new ort.Tensor('float32', float32Data, [1, 3, 640, 640]);
    return tensor;
  }
  
  postProcess(results) {
    // YOLOv8 output format: [1, 84, 8400]
    // 84 = 4 bbox coords + 80 class scores (COCO) or your custom classes
    // 8400 = number of anchor points
    
    const output = results.output0.data;
    const detections = [];
    const confidenceThreshold = 0.45;  // Adjust as needed
    const iouThreshold = 0.45;
    
    // Parse detections
    for (let i = 0; i < 8400; i++) {
      // Extract bbox and scores
      const x = output[i];
      const y = output[8400 + i];
      const w = output[16800 + i];
      const h = output[25200 + i];
      
      // Find max class score
      let maxScore = 0;
      let maxClass = 0;
      for (let c = 0; c < this.numClasses; c++) {
        const score = output[(4 + c) * 8400 + i];
        if (score > maxScore) {
          maxScore = score;
          maxClass = c;
        }
      }
      
      if (maxScore > confidenceThreshold) {
        detections.push({
          x: x,
          y: y,
          width: w,
          height: h,
          confidence: maxScore,
          classId: maxClass,
          className: this.classNames[maxClass]
        });
      }
    }
    
    // Apply Non-Maximum Suppression
    const finalDetections = this.applyNMS(detections, iouThreshold);
    
    return finalDetections;
  }
  
  applyNMS(detections, iouThreshold) {
    // Sort by confidence
    detections.sort((a, b) => b.confidence - a.confidence);
    
    const keep = [];
    const suppressed = new Set();
    
    for (let i = 0; i < detections.length; i++) {
      if (suppressed.has(i)) continue;
      
      keep.push(detections[i]);
      
      for (let j = i + 1; j < detections.length; j++) {
        if (suppressed.has(j)) continue;
        
        const iou = this.calculateIOU(detections[i], detections[j]);
        if (iou > iouThreshold) {
          suppressed.add(j);
        }
      }
    }
    
    return keep;
  }
  
  calculateIOU(box1, box2) {
    const x1_min = box1.x - box1.width / 2;
    const y1_min = box1.y - box1.height / 2;
    const x1_max = box1.x + box1.width / 2;
    const y1_max = box1.y + box1.height / 2;
    
    const x2_min = box2.x - box2.width / 2;
    const y2_min = box2.y - box2.height / 2;
    const x2_max = box2.x + box2.width / 2;
    const y2_max = box2.y + box2.height / 2;
    
    const intersect_x_min = Math.max(x1_min, x2_min);
    const intersect_y_min = Math.max(y1_min, y2_min);
    const intersect_x_max = Math.min(x1_max, x2_max);
    const intersect_y_max = Math.min(y1_max, y2_max);
    
    const intersect_area = Math.max(0, intersect_x_max - intersect_x_min) *
                          Math.max(0, intersect_y_max - intersect_y_min);
    
    const box1_area = box1.width * box1.height;
    const box2_area = box2.width * box2.height;
    const union_area = box1_area + box2_area - intersect_area;
    
    return intersect_area / union_area;
  }
  
  loadImage(imageData) {
    return new Promise((resolve) => {
      const img = new Image();
      img.onload = () => resolve(img);
      
      if (imageData instanceof Blob) {
        img.src = URL.createObjectURL(imageData);
      } else {
        img.src = imageData;
      }
    });
  }
}

Field Testing Protocol
javascriptclass FieldTester {
  constructor(detector, groundTruth) {
    this.detector = detector;
    this.groundTruth = groundTruth;  // Manually verified detections
    this.results = {
      truePositives: 0,
      falsePositives: 0,
      falseNegatives: 0,
      detectionTimes: []
    };
  }
  
  async testRoute(videoStream) {
    // Run detection on recorded route
    // Compare with ground truth annotations
    
    for (const frame of videoStream) {
      const startTime = performance.now();
      const detections = await this.detector.detect(frame);
      const detectionTime = performance.now() - startTime;
      
      this.results.detectionTimes.push(detectionTime);
      
      // Compare with ground truth
      this.compareWithGroundTruth(detections, frame.groundTruth);
    }
    
    return this.calculateMetrics();
  }
  
  calculateMetrics() {
    const precision = this.results.truePositives / 
                     (this.results.truePositives + this.results.falsePositives);
    const recall = this.results.truePositives / 
                  (this.results.truePositives + this.results.falseNegatives);
    const f1Score = 2 * (precision * recall) / (precision + recall);
    
    const avgDetectionTime = this.results.detectionTimes.reduce((a, b) => a + b, 0) / 
                            this.results.detectionTimes.length;
    
    return {
      precision: precision,
      recall: recall,
      f1Score: f1Score,
      avgDetectionTime: avgDetectionTime,
      truePositives: this.results.truePositives,
      falsePositives: this.results.falsePositives,
      falseNegatives: this.results.falseNegatives
    };
  }
}

            resume=True,  # Continue from previous weights

    def evaluate_improvement(self):

        }

class ActiveLearner {
  async flagUncertainPredictions(detection) {
    const uncertainty = this.calculateUncertainty(detection);
    
    if (uncertainty > threshold) {
      // Save for manual review and retraining
      await this.saveForReview(detection);
    }
  }
  
  calculateUncertainty(detection) {
    // Low confidence predictions
    if (detection.confidence < 0.65) return 'low_confidence';
    
    // Multiple similar detections (confusion)
    if (this.hasMultipleSimilarClasses(detection)) return 'class_confusion';
    
    // Object at edge of frame
    if (this.isAtEdge(detection.bbox)) return 'edge_case';
    
    return 'confident';
  }
