rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function isAdmin() {
      return request.auth != null && request.auth.token.admin == true;
    }
    
    function isOwner(userId) {
      return request.auth != null && request.auth.uid == userId;
    }
    
    // License Features - Read by all authenticated, write only via Cloud Functions
    match /licenseFeatures/{featureId} {
      allow read: if isAuthenticated();
      allow write: if false; // Only via Cloud Functions
    }
    
    // License Packages - Read by all authenticated, write only via Cloud Functions
    match /licensePackages/{packageId} {
      allow read: if isAuthenticated();
      allow write: if false; // Only via Cloud Functions
    }
    
    // Activation Codes - NO direct access, only via Cloud Functions
    match /activationCodes/{codeId} {
      allow read, write: if false; // Only via Cloud Functions
    }
    
    // User Licenses - Users can read their own, write only via Cloud Functions
    match /userLicenses/{licenseId} {
      allow read: if isAuthenticated() && 
                   (isAdmin() || resource.data.userId == request.auth.uid);
      allow write: if false; // Only via Cloud Functions
    }
    
    // User Devices - Users can read their own, write only via Cloud Functions
    match /userDevices/{deviceId} {
      allow read: if isAuthenticated() && 
                   (isAdmin() || resource.data.userId == request.auth.uid);
      allow write: if false; // Only via Cloud Functions
    }
    
    // App-specific collections (not licensing-related)
    // Accounts - User registration and profiles
    match /accounts/{accountId} {
      allow read: if isAuthenticated();
      allow create: if true; // Allow unauthenticated users to register
      allow update, delete: if isAuthenticated() && 
                             (isAdmin() || resource.data.uid == request.auth.uid);
    }
    
    // Surveys - Users can manage their own surveys
    // Security model: ownerId set on create, verified on update/delete
    match /surveys/{surveyId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && request.resource.data.ownerId == request.auth.uid;
      allow update, delete: if isAuthenticated() && 
                             (isAdmin() || resource.data.ownerId == request.auth.uid);
    }
    
    // POIs (Points of Interest) - Users can manage their own POIs
    match /pois/{poiId} {
      allow read: if isAuthenticated();
      allow create, update, delete: if isAuthenticated();
    }
    
    // Measurements - Users can manage their own measurements
    // Security model: ownerId set on create, verified on update/delete
    match /measurements/{measurementId} {
      allow read: if isAuthenticated();
      allow create: if isAuthenticated() && request.resource.data.ownerId == request.auth.uid;
      allow update, delete: if isAuthenticated() && 
                             (isAdmin() || resource.data.ownerId == request.auth.uid);
    }
    
     // Live Monitor Feeds - authenticated users can read/write their own data
    match /liveMonitorFeeds/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    // User Email Mapping - authenticated users can read all, write their own
    match /userEmailMapping/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == userId;
    }
    
    
    // IMPORTANT: No catch-all rule - all collections must be explicitly defined
    // This ensures licensing collections remain protected and only accessible via Cloud Functions
  }
}
