## 1. CONTEXT ABOUT THE LASER PROTOCOL (IMPORTANT)

We have an existing Windows C# app “RSA Vertical Clearance Measurement App” that already works with this laser. You must **replicate its protocol** in MeasurePRO.

### Serial port settings

The laser must be opened with:

- Port: selected by user (COMx) – don’t hard-code.
- Baud rate: **460800**
- Data bits: **8**
- Stop bits: **1**
- Parity: **none** (8N1)
- No hardware / software flow control.

### Control commands

The laser uses simple ASCII commands:

- **Start streaming:** send the ASCII string `DT` followed by carriage return:

  - Command: `"DT\r"`

- **Stop streaming:** send ESC (0x1B):

  - Command: `"\x1B"`

No other commands are required for the first version.

### Incoming data format (laser → PC)

Once streaming is started, the laser sends a **continuous binary stream** grouped conceptually as **3-byte frames** with this pattern:

- Frame = `[b0, b1, b2]` (3 bytes, no delimiter).
- Valid frame condition (framing rule):

  - `b0 & 0x80 != 0`  (MSB of byte 0 is **1**)
  - `b1 & 0x80 == 0`  (MSB of byte 1 is **0**)
  - `b2 & 0x80 == 0`  (MSB of byte 2 is **0**)

So valid frames look like:

- `1xxxxxxx 0xxxxxxx 0xxxxxxx`

If the pattern is broken, you must discard bytes and resync.

### Distance / intensity decoding

For a valid frame `[b0, b1, b2]`:

- **Distance:**

  ```ts
  const raw = (((b0 & 0x7f) << 7) | b1);  // 14-bit value
  const distanceM = raw / 100.0;          // meters, 0.01 m resolution
Typical range of interest: 0–30 m (even though encoding allows up to 163.83 m).

Intensity / quality:

ts
Copy code
const intensity = b2 * 2; // 0–510
You must convert this into a normalized measurement object consumed by MeasurePRO.

2.2. Implement RSA driver
Create src/hardware/laser/rsaThreeByteDriver.ts (or equivalent) and implement a driver class that matches the C# behavior.

Key requirements:

Maintain an internal byte buffer.

Try to parse frames of 3 bytes.

Use the MSB pattern to identify valid frames.

Apply optional range-gating (e.g., 0–30 m).

Example implementation (you can refine / optimize):

ts
Copy code
import {
  LaserDriver,
  LaserProfileConfig,
  NormalizedLaserMeasurement
} from './types';

export class RsaThreeByteLaserDriver implements LaserDriver {
  public readonly profile: LaserProfileConfig;
  private buffer: number[] = [];

  constructor(profile: LaserProfileConfig) {
    this.profile = profile;
  }

  async onOpen(_port: SerialPort) {
    // Nothing special yet, but keep this hook for future DTR handling etc.
  }

  async startStreaming(port: SerialPort) {
    const cmd = this.profile.startCommand ?? 'DT\r';
    await this.writeAscii(port, cmd);
  }

  async stopStreaming(port: SerialPort) {
    const cmd = this.profile.stopCommand ?? '\x1B';
    await this.writeAscii(port, cmd);
  }

  feedBytes(chunk: Uint8Array): NormalizedLaserMeasurement[] {
    const results: NormalizedLaserMeasurement[] = [];
    const now = Date.now();

    // append new bytes
    for (let i = 0; i < chunk.length; i++) {
      this.buffer.push(chunk[i]);
    }

    // process while at least 3 bytes
    while (this.buffer.length >= 3) {
      const b0 = this.buffer[0];
      const b1 = this.buffer[1];
      const b2 = this.buffer[2];

      const validFrame =
        (b0 & 0x80) !== 0 &&
        (b1 & 0x80) === 0 &&
        (b2 & 0x80) === 0;

      if (validFrame) {
        const raw = (((b0 & 0x7f) << 7) | b1);
        const distanceM = raw / 100.0;
        const intensity = b2 * 2;

        const maxRange = this.profile.options?.maxRangeM ?? 30;
        const inRange =
          distanceM > 0 &&
          (maxRange <= 0 || distanceM <= maxRange);

        results.push({
          distanceM: inRange ? distanceM : null,
          intensity,
          timestamp: now,
          raw: new Uint8Array([b0, b1, b2])
        });

        // consume processed frame
        this.buffer.splice(0, 3);
      } else {
        // Failed framing, drop 1 byte and retry (simple resync).
        this.buffer.shift();
      }
    }

    return results;
  }

  private async writeAscii(port: SerialPort, text: string) {
    const writer = port.writable?.getWriter();
    if (!writer) return;
    const enc = new TextEncoder();
    await writer.write(enc.encode(text));
    writer.releaseLock();
  }
}

ts
Copy code
import {
  LaserProfileConfig,
  NormalizedLaserMeasurement
} from './types';
import { createLaserDriver } from './types'; // adjust import to where the factory lives
import { RsaThreeByteLaserDriver } from './rsaThreeByteDriver'; // if needed

type MeasurementListener = (m: NormalizedLaserMeasurement) => void;

export class LaserService {
  private port: SerialPort | null = null;
  private reader: ReadableStreamDefaultReader<Uint8Array> | null = null;
  private profile: LaserProfileConfig;
  private driver = createLaserDriver;
  private listeners = new Set<MeasurementListener>();
  private running = false;

  constructor(profile: LaserProfileConfig) {
    this.profile = profile;
    this.driver = createLaserDriver(profile);
  }

  addListener(fn: MeasurementListener) {
    this.listeners.add(fn);
  }

  removeListener(fn: MeasurementListener) {
    this.listeners.delete(fn);
  }

  private emit(measurement: NormalizedLaserMeasurement) {
    for (const fn of this.listeners) {
      fn(measurement);
    }
  }

  async connectAndStart() {
    if (!('serial' in navigator)) {
      throw new Error('Web Serial API not available in this browser.');
    }

    this.port = await (navigator as any).serial.requestPort();
    await this.port.open({
      baudRate: this.profile.baudRate,
      dataBits: this.profile.dataBits,
      stopBits: this.profile.stopBits,
      parity: this.profile.parity
    });

    await this.driver.onOpen?.(this.port);

    // Start streaming
    await this.driver.startStreaming(this.port);

    this.running = true;
    this.reader = this.port.readable!.getReader();

    // Read loop
    (async () => {
      while (this.running && this.reader) {
        const { value, done } = await this.reader.read();
        if (done || !value) break;
        const measurements = this.driver.feedBytes(value);
        for (const m of measurements) {
          this.emit(m);
        }
      }
    })().catch(err => {
      console.error('Laser read loop error', err);
    });
  }

  async stop() {
    this.running = false;

    if (this.reader) {
      await this.reader.cancel().catch(() => {});
      this.reader.releaseLock();
      this.reader = null;
    }

    if (this.port) {
      try {
        await this.driver.stopStreaming(this.port);
      } catch (err) {
        console.warn('Error stopping laser stream', err);
      }
      await this.port.close();
      this.port = null;
    }
  }
}

export function useLaserMeasurements(profile: LaserProfileConfig | null) {
  const [lastMeasurement, setLastMeasurement] =
    useState<NormalizedLaserMeasurement | null>(null);
  const [error, setError] = useState<Error | null>(null);
  const [connected, setConnected] = useState(false);

  useEffect(() => {
    if (!profile) return;
    const service = new LaserService(profile);
    service.addListener(m => setLastMeasurement(m));
    service.connectAndStart()
      .then(() => setConnected(true))
      .catch(err => setError(err));

    return () => {
      service.stop().catch(() => {});
    };
  }, [profile?.id]);

  return { lastMeasurement, error, connected };
}
